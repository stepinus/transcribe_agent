        –≠—Ç–æ—Ç –º–µ—Ç–æ–¥ –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å –≤—ã–∑–≤–∞–Ω –ø–æ—Å–ª–µ —Å–æ–∑–¥–∞–Ω–∏—è —ç–∫–∑–µ–º–ø–ª—è—Ä–∞ AllTalkAPI.
        –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç True –µ—Å–ª–∏ –∏–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è —É—Å–ø–µ—à–Ω–∞, False –≤ –ø—Ä–æ—Ç–∏–≤–Ω–æ–º —Å–ª—É—á–∞–µ.

        self.available_voices = self.get_available_voices()
        self.available_rvc_voices = self.get_available_rvc_voices()
        return True

    def reload_config(self):
        """
        –ü–µ—Ä–µ–∑–∞–≥—Ä—É–∑–∫–∞ –∫–æ–Ω—Ñ–∏–≥—É—Ä–∞—Ü–∏–∏ —Å–µ—Ä–≤–µ—Ä–∞ AllTalk.

            return True
        return False

    def generate_tts(self, text, character_voice, narrator_voice=None, **kwargs):
        """
        –ì–µ–Ω–µ—Ä–∞—Ü–∏—è –∞—É–¥–∏–æ –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏—è —Ç–µ–∫—Å—Ç–∞ –≤ —Ä–µ—á—å –∏—Å–ø–æ–ª—å–∑—É—è —Å–µ—Ä–≤–µ—Ä AllTalk.
        
        Args:
            text (str): –¢–µ–∫—Å—Ç –¥–ª—è –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏—è –≤ —Ä–µ—á—å.
            character_voice (str): –ì–æ–ª–æ—Å –¥–ª—è –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è –ø–µ—Ä—Å–æ–Ω–∞–∂–µ–º.
            narrator_voice (str, optional): –ì–æ–ª–æ—Å –¥–ª—è –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è —Ä–∞—Å—Å–∫–∞–∑—á–∏–∫–æ–º, –µ—Å–ª–∏ –ø—Ä–∏–º–µ–Ω–∏–º–æ.
            **kwargs: –î–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω—ã–µ –ø–∞—Ä–∞–º–µ—Ç—Ä—ã –¥–ª—è –≥–µ–Ω–µ—Ä–∞—Ü–∏–∏ TTS (–Ω–∞–ø—Ä–∏–º–µ—Ä, language, output_file_name).
        
        Returns:
            dict: –°–ª–æ–≤–∞—Ä—å —Å–æ–¥–µ—Ä–∂–∞—â–∏–π –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏—é –æ —Å–≥–µ–Ω–µ—Ä–∏—Ä–æ–≤–∞–Ω–Ω–æ–º –∞—É–¥–∏–æ, –∏–ª–∏ None –µ—Å–ª–∏ –≥–µ–Ω–µ—Ä–∞—Ü–∏—è –Ω–µ—É–¥–∞—á–Ω–∞.
        """
        data = {
            "text_input": text,
            "character_voice_gen": character_voice,
            "narrator_enabled": "true" if narrator_voice else "false",
            "narrator_voice_gen": narrator_voice,
            **kwargs
        }
        response = requests.post(f"{self.base_url}/api/tts-generate", data=data)
        return response.json() if response.status_code == 200 else None

    def switch_model(self, model_name):
        """
        –ü–µ—Ä–µ–∫–ª—é—á–µ–Ω–∏–µ –Ω–∞ –¥—Ä—É–≥—É—é TTS –º–æ–¥–µ–ª—å.
        
        Args:
            model_name (str): –ò–º—è –º–æ–¥–µ–ª–∏ –¥–ª—è –ø–µ—Ä–µ–∫–ª—é—á–µ–Ω–∏—è.
        
        Returns:

        """
        try:
            response = requests.post(f"{self.base_url}/api/reload", params={"tts_method": model_name})
            response.raise_for_status()  # –≠—Ç–æ –≤—ã–∑–æ–≤–µ—Ç –∏—Å–∫–ª—é—á–µ–Ω–∏–µ –¥–ª—è HTTP –æ—à–∏–±–æ–∫

            print(f"–û—à–∏–±–∫–∞ –ø–µ—Ä–µ–∫–ª—é—á–µ–Ω–∏—è –º–æ–¥–µ–ª–∏: {e}")
            if response.status_code == 404:
                print(f"–ú–æ–¥–µ–ª—å '{model_name}' –Ω–µ –Ω–∞–π–¥–µ–Ω–∞ –Ω–∞ —Å–µ—Ä–≤–µ—Ä–µ.")
            elif response.status_code == 500:
                print("–°–µ—Ä–≤–µ—Ä —Å—Ç–æ–ª–∫–Ω—É–ª—Å—è —Å –æ—à–∏–±–∫–æ–π –ø—Ä–∏ –ø–µ—Ä–µ–∫–ª—é—á–µ–Ω–∏–∏ –º–æ–¥–µ–ª–µ–π.")
